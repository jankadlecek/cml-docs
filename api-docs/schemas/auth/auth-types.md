# Auth Configuration Types

> **Context for AI:** These schemas define how CML connects to external services.
> Each auth type represents a different authentication method:
> - **ApiKey**: Simple API key authentication
> - **BasicToBearer**: Basic auth that exchanges for a bearer token
> - **AzureServicePrincipal**: Azure AD service principal authentication
> - **Credentials**: Username/password authentication
> - **CustomAuthHandler**: Custom authentication logic
> - **GoogleCredentials**: Google service account credentials
>
> Auth configs are stored via the Components API and referenced in input/output clients.
>
> **See also:** [Components endpoints](../../endpoints/components.md), [Connections](../connections.md)

### `IBasicToBearerAuthData`

Provides definition to handle basic authorization to generate bearer token.

```json
{
  "description": "Provides definition to handle basic authorization to generate bearer token.",
  "properties": {
    "contentType": {
      "type": "string",
      "enum": [
        "X_FROM_URLENCODED",
        "JSON"
      ],
      "description": "Request content type to be used. If not preset default content type (application/json) is used.",
      "example": "X_FROM_URLENCODED"
    },
    "headers": {
      "$ref": "#/components/schemas/AdditionalProperties",
      "description": "Headers to be sent in authorization request."
    },
    "data": {
      "$ref": "#/components/schemas/AdditionalProperties",
      "description": "Data to be sent in authorization request."
    },
    "method": {
      "type": "string",
      "description": "Http method.",
      "example": "POST | PUT | GET"
    },
    "responseHandler": {
      "properties": {
        "expires": {
          "anyOf": [
            {
              "properties": {
                "unit": {
                  "$ref": "#/components/schemas/TimeUnitEnum",
                  "description": "The unit of given value.\nDefault value is `ms` but some APIs use seconds `s`.",
                  "example": "s"
                },
                "defaultValue": {
                  "type": "number",
                  "format": "double",
                  "description": "Optional param used as default value if expiration param is not in the response. This value should contain number of s (ms) to token expiration limit."
                },
                "paramName": {
                  "type": "string",
                  "nullable": true,
                  "description": "Expiration`s param name in response.",
                  "example": "expires_in"
                }
              },
              "required": [
                "paramName"
              ],
              "type": "object"
            },
            {
              "type": "string",
              "enum": [
                "UNLIMITED"
              ]
            }
          ],
          "description": "Define how to handle token expiration.\n\nIf:\n- not present token is generated for every call\n- expires = 'UNLIMITED' -> token is generated ones\n- expires = object -> token is generated by given expiration time"
        },
        "tokenParamName": {
          "type": "string",
          "description": "Attribute name where token is stored in response.",
          "example": "token | access_token"
        }
      },
      "required": [
        "tokenParamName"
      ],
      "type": "object",
      "description": "Defines how to map response to our object."
    },
    "url": {
      "type": "string",
      "description": "Authentication url.",
      "example": "{{cml_url}}/api/rest/v1/users/authenticate"
    }
  },
  "required": [
    "data",
    "method",
    "responseHandler",
    "url"
  ],
  "type": "object",
  "additionalProperties": false
}
```

### `IAzureServicePrincipalAuthData`

Provides definition to handle Service Principal Client authorization for communication with Azure. To create new Service Principal please refer to
[Azure](https://portal.azure.com/#view/Microsoft_AAD_IAM/ActiveDirectoryMenuBlade/~/RegisteredApps)

```json
{
  "description": "Provides definition to handle Service Principal Client authorization for communication with Azure. To create new Service Principal please refer to\n[Azure](https://portal.azure.com/#view/Microsoft_AAD_IAM/ActiveDirectoryMenuBlade/~/RegisteredApps)",
  "properties": {
    "tenantId": {
      "type": "string",
      "description": "Required param. Tenant ID in the [Azure](https://portal.azure.com/#view/Microsoft_AAD_IAM/ActiveDirectoryMenuBlade/~/Overview).",
      "example": "68bb8912-8677-4739-b182-cda507674321"
    },
    "clientId": {
      "type": "string",
      "description": "Required param. Service Principal Client ID.",
      "example": "68bb8912-8677-4739-b182-cda507123456"
    },
    "clientSecret": {
      "type": "string",
      "description": "Required param. Service Principal Client secret.",
      "example": "client-secret"
    },
    "scopes": {
      "items": {
        "type": "string"
      },
      "type": "array",
      "description": "Required param. How to define scopes please refer to [MS Web](https://learn.microsoft.com/en-us/azure/active-directory/develop/scopes-oidc)"
    },
    "authority": {
      "type": "string",
      "description": "Optional param. Defines which authority should be used.\n\nDefault value: https://login.microsoftonline.com",
      "example": "https://login.microsoftonline.com"
    }
  },
  "required": [
    "tenantId",
    "clientId",
    "clientSecret",
    "scopes"
  ],
  "type": "object",
  "additionalProperties": false
}
```

{
  "properties": {
    "name": {
